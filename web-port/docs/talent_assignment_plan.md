## 📝 구현 계획

### 1단계: 특성 ID 상수 정의
**파일**: `src/utils/characterFactory.ts`

```typescript
// 초기 부여 가능한 특성 ID 목록
const ASSIGNABLE_TALENTS = {
  personality: [10, 11, 12, 14, 15, 16, 17, 18],
  sexualInterest: [20, 21, 23, 24, 25, 27, 28],
  virginity: [30, 32, 33, 35, 36, 37],
  constitution: [40, 41, 43, 44, 45],
  skills: [50, 51, 55, 56, 60, 61, 62, 63, 64],
  honesty: [70, 71, 72, 73, 79, 80],
  perversion: [82, 84, 90, 91, 92, 93],
  body: [
    101, 102, 103, 104, 105, 106, 107, 108,
    111, 112, 113, 117, 118, 119,
    121, 122, 124, 125, 126, 127, 128,
    132, 133, 134, 135,
    140, 141,
    150, 151, 152
  ]
};

// 서로 배타적인 특성 쌍
const EXCLUSIVE_PAIRS = [
  [11, 14],   // 건방짐 vs 얌전함
  [15, 17],   // 프라이드 높음 vs 낮음
  [20, 21],   // 자제심 vs 쿨 (같이 가능하지만 둘 다 상실되므로 한 개만)
  [23, 24],   // 호기심 vs 보수적
  [25, 26],   // 포지티브 vs 네거티브 (26은 부여 금지)
  [30, 31],   // 정조관념 vs 무관심 (31은 부여 금지)
  [32, 33],   // 억압 vs 해방
  [35, 36],   // 수줍음 vs 부끄럼 없음
  [40, 41],   // 아픔에 약함 vs 강함
  [42, 43],   // 젖기쉬움 vs 힘듬 (42는 부여 금지)
  [44, 45],   // 울보 vs 울지 않음
  [50, 51],   // 소질있음 vs 습득느림
  [61, 62],   // 악취둔감 vs 민감
  [70, 71],   // 쾌감에 솔직 vs 부정
  [101, 102], // C둔감 vs 민감
  [103, 104], // V둔감 vs 민감
  [105, 106], // A둔감 vs 민감
  [107, 108], // B둔감 vs 민감
  [109, 110, 114, 116], // 빈유/거유/폭유/절벽 (4-way exclusive, 자동 설정)
  [111, 112], // 회복 빠름 vs 느림
  [125, 128], // 음모 없음 vs 빽빽함
  [140, 141], // 동물 좋아함 vs 싫어함
];
```

### 2단계: assignWithChance 수정
```typescript
const assignWithChance = (
  allowedIds: number[],
  excludeIds: number[] = [],
  skipExclusivityCheck = false
) => {
  if (chance(0.25)) {
    const available = allowedIds.filter(id => {
      if (excludeIds.includes(id) || talents[id]) return false;

      // 배타적 쌍 체크
      if (!skipExclusivityCheck) {
        for (const pair of EXCLUSIVE_PAIRS) {
          if (pair.includes(id)) {
            // 이 쌍의 다른 특성이 이미 있으면 제외
            const hasOther = pair.some(otherId =>
              otherId !== id && talents[otherId]
            );
            if (hasOther) return false;
          }
        }
      }

      return true;
    });

    if (available.length > 0) {
      const selectedId = randomPick(available);
      talents[selectedId] = 1;
      excludeIds.push(selectedId);

      // 재귀: 다시 1/4 확률로 추가 시도
      assignWithChance(allowedIds, excludeIds, skipExclusivityCheck);
    }
  }
};
```

### 3단계: generateRandomTalents 재작성
```typescript
function generateRandomTalents(config: TalentConfig): CharacterTalents {
  const { age, isVirgin, breastSize } = config;
  const talents: CharacterTalents = {};

  // 1. 필수 특성 (처녀성)
  if (isVirgin) {
    talents[0] = 1;  // 처녀
    talents[2] = 1;  // 애널처녀
  }
  if (!chance(0.7)) talents[3] = 1; // 순결 (30%)

  // 2. 성격 (1/4 확률로 추가)
  assignWithChance(ASSIGNABLE_TALENTS.personality);

  // 3. 성적 관심 (1/4 확률로 추가)
  assignWithChance(ASSIGNABLE_TALENTS.sexualInterest);

  // 4. 처녀성 관련 (처녀면만, 1/4 확률로 추가)
  if (isVirgin) {
    assignWithChance(ASSIGNABLE_TALENTS.virginity);
  }

  // 5. 체질 (1/4 확률로 추가)
  assignWithChance(ASSIGNABLE_TALENTS.constitution);

  // 6. 기술/재능 (1/4 확률로 추가)
  assignWithChance(ASSIGNABLE_TALENTS.skills);

  // 7. 솔직도 (1/4 확률로 추가)
  assignWithChance(ASSIGNABLE_TALENTS.honesty);

  // 8. 성벽/기타 (1/4 확률로 추가)
  assignWithChance(ASSIGNABLE_TALENTS.perversion);

  // 9. 신체 특징 - 자동 설정
  // 키 기반
  if (config.height > 170) talents[99] = 1;  // 큰체구
  if (config.height < 150) talents[100] = 1; // 작은체형

  // 가슴 크기 (이미 구현됨)
  if (breastSize === 1) talents[109] = 1;        // 빈유
  else if (breastSize === 2) talents[116] = 1;   // 절벽
  else if (breastSize >= 6) talents[110] = 1;    // 거유
  else if (breastSize >= 7) talents[114] = 1;    // 폭유

  // 10. 기타 신체 특징 (1/4 확률로 추가)
  assignWithChance(ASSIGNABLE_TALENTS.body);

  return talents;
}
```

## ✅ 검증 체크리스트

1. **획득계 특성 미부여**: 74-78, 83, 85-89, 130-137, 230-233, 271-272, 430, 440 등
2. **약물/정신 특성 미부여**: 9, 22, 26, 46, 123
3. **이벤트 특성 미부여**: 57, 115, 131
4. **배타적 쌍 체크**: 같은 쌍의 특성이 동시에 부여되지 않음
5. **처녀성 일관성**: 처녀면 V경험 0, 처녀 관련 특성만 부여
6. **신체 자동 설정**: 키/가슴 기반 특성이 올바르게 설정됨
